### 1. **ChainContext __post_init__ Violates Frozen Dataclass**

### 2. **API Key Verification Has TOCTOU Race Condition**
```python
async def verify_api_key(...) -> APIKey:
    async with db_pool.acquire() as conn:
        row = await conn.fetchrow(...)  # Check key
        
        if row:
            await conn.execute(
                "UPDATE api_keys SET last_used_at = NOW() WHERE id = $1",
                row['id']
            )  # Update timestamp - separate query!
    
    if not row:  # Check happens AFTER release of connection
        raise HTTPException(...)
```

**Problem**: Between fetchrow and the final check, another request could revoke the key
**Fix**: Use single transaction
```python
async def verify_api_key(
    api_key: Optional[str] = Security(api_key_header),
    db_pool: asyncpg.Pool = Depends(get_db_pool)
) -> APIKey:
    if not api_key:
        raise HTTPException(status_code=401, detail="API key required")
    
    key_hash = hashlib.sha256(api_key.encode()).hexdigest()
    
    async with db_pool.acquire() as conn:
        async with conn.transaction():
            # Atomic check and update
            row = await conn.fetchrow(
                """
                UPDATE api_keys 
                SET last_used_at = NOW() 
                WHERE key_hash = $1 
                  AND is_active = true
                  AND (expires_at IS NULL OR expires_at > NOW())
                RETURNING id, user_id, rate_limit, expires_at
                """,
                key_hash
            )
    
    if not row:
        raise HTTPException(status_code=403, detail="Invalid or expired API key")
    
    return APIKey(
        id=str(row['id']),
        user_id=row['user_id'],
        rate_limit=row['rate_limit'],
        expires_at=row['expires_at']
    )
```

### 8. **Ground Truth Agreement Calculation Flawed**

```python
def _calculate_agreement(
    judgments_1: Dict[str, int], 
    judgments_2: Dict[str, int]
) -> float:
    """
    Calculate agreement using Cohen's Kappa (simplified).
    
    Only considers documents that were judged by at least one LLM.
    Ignores documents neither LLM mentioned (not truly "agreement").
    """
    all_docs = set(judgments_1.keys()) | set(judgments_2.keys())
    
    if not all_docs:
        return 1.0  # No judgments to compare
    
    # Observed agreement
    agreements = sum(
        1 for doc in all_docs 
        if judgments_1.get(doc, 0) == judgments_2.get(doc, 0)
    )
    po = agreements / len(all_docs)
    
    # Expected agreement by chance
    # (simplified - true Cohen's Kappa more complex)
    scores = [judgments_1.get(doc, 0) for doc in all_docs]
    scores += [judgments_2.get(doc, 0) for doc in all_docs]
    
    # Proportion of each rating
    from collections import Counter
    counts = Counter(scores)
    total = len(scores)
    pe = sum((count / total) ** 2 for count in counts.values())
    
    # Cohen's Kappa
    if pe == 1.0:
        return 1.0
    
    kappa = (po - pe) / (1 - pe)
    return max(0.0, kappa)  # Clip to [0, 1]
```

```python
@classmethod
def from_dataset(
    cls, 
    dataset: 'TestDataset', 
    test_ratio: float = 0.2,
    random_seed: int = 42,
    stratify_by: Optional[str] = None
) -> 'DatasetSplit':
    """
    Create stratified train/test split.
    
    Args:
        stratify_by: Metadata field to stratify by (e.g., 'query_type')
    """
    random.seed(random_seed)
    
    if stratify_by is None:
        # Simple random split
        queries = dataset.queries.copy()
        random.shuffle(queries)
        split_idx = int(len(queries) * (1 - test_ratio))
        return cls(train=queries[:split_idx], test=queries[split_idx:])
    
    # Stratified split
    from collections import defaultdict
    
    # Group by stratification field
    groups = defaultdict(list)
    for query in dataset.queries:
        key = query.metadata.get(stratify_by, 'unknown')
        groups[key].append(query)
    
    train, test = [], []
    for group_queries in groups.values():
        random.shuffle(group_queries)
        split_idx = int(len(group_queries) * (1 - test_ratio))
        train.extend(group_queries[:split_idx])
        test.extend(group_queries[split_idx:])
    
    return cls(train=train, test=test)
```

```python
@classmethod
def register(
    cls, 
    name: str,
    description: str = "",
    resources: 'StrategyResources' = None,
    override: bool = False
):
    """
    Decorator to register strategies.
    
    Args:
        override: Allow overriding existing strategy (default: False)
    """
    def decorator(strategy_class: Type[StrategyProtocol]):
        # Validate name
        if not name or not name.replace('_', '').isalnum():
            raise ValueError(
                f"Invalid strategy name '{name}'. "
                "Must be alphanumeric with underscores."
            )
        
        # Check for duplicates
        if name in cls._strategies and not override:
            raise ValueError(
                f"Strategy '{name}' already registered. "
                f"Use override=True to replace."
            )
        
        # Validate protocol compliance
        if not hasattr(strategy_class, 'execute'):
            raise TypeError(
                f"Strategy class {strategy_class.__name__} "
                "must implement execute() method"
            )
        
        cls._strategies[name] = strategy_class
        cls._metadata[name] = {
            "description": description,
            "resources": resources or StrategyResources(),
            "class": strategy_class.__name__
        }
        
        logger.info(f"Registered strategy: {name}")
        return strategy_class
    
    return decorator
```
